    <Colorize color="somcheat-blue" is_both={true}>Variables</Colorize> waxa loo adeegsada in xog lagu kaydiyo.
      <Colorize color="somcheat-gray" is_both={true}>Qaabka Isticmaalka (Syntax):</Colorize>
      <Code>
          ```js
          <keyword> <variable_name> = <initial_value>;
          ```
      </Code>
      <Colorize color="somcheat-gray" is_both={true}>Tusaale:</Colorize>
      <Code>
          ```js
          var name = "Adam";
          let age = 25;
          const gender = "male";
          
          console.log(name, age, gender);
          ```
      </Code>
      <Colorize color="somcheat-gray" is_both={true}>Natiijada:</Colorize>
      <Terminal>
          ```text
          Adam 25 male
          ```
      </Terminal>
      <Message type="note">

      ```js
      let name = "Elmi";
      console.log(name);
      ```
